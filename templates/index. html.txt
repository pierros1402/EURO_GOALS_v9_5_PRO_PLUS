  <script>
    (function () {
      const AISCORE_BASE = "https://www.aiscore.com/";

      const state = {
        view: "live",      // live | scores | standings | leagues | teams | ai | goalmatrix | smartmoney | about
        theme: "dark",     // dark | blue | white
      };

      const modeLabel = document.getElementById("mode-label");
      const modeThemeLabel = document.getElementById("mode-theme-label");
      const iframe = document.getElementById("aiscore-iframe");
      const iframeWrapper = document.getElementById("iframe-wrapper");

      const widgetTitleLabel = document.getElementById("widget-title-label");
      const widgetSubtitleLabel = document.getElementById("widget-subtitle-label");

      const primaryNav = document.getElementById("primary-nav");
      const aiNav = document.getElementById("ai-nav");
      const themeButtons = document.querySelectorAll(".theme-btn");

      const paneAi = document.getElementById("pane-ai");
      const paneGoal = document.getElementById("pane-goalmatrix");
      const paneSmart = document.getElementById("pane-smartmoney");
      const paneAbout = document.getElementById("pane-about");
      const panes = [paneAi, paneGoal, paneSmart, paneAbout];

      function loadPersistedTheme() {
        try {
          const saved = localStorage.getItem("aimatchlab_theme");
          if (saved === "dark" || saved === "blue" || saved === "white") {
            state.theme = saved;
          }
        } catch (e) {
          // ignore
        }
      }

      function persistTheme() {
        try {
          localStorage.setItem("aimatchlab_theme", state.theme);
        } catch (e) {
          // ignore
        }
      }

      function themeLabelName(theme) {
        if (theme === "blue") return "Blue";
        if (theme === "white") return "Light";
        return "Dark";
      }

      function buildAiscoreUrl(view, theme) {
        // AiScore δεν έχει public documented deep-links για tabs,
        // οπότε χρησιμοποιούμε τη βασική σελίδα με theme & responsive width.
        const params = new URLSearchParams();
        params.set("width", "100%");
        params.set("theme", theme);
        // Αν κάποια στιγμή προστεθεί anchor για live/results κτλ,
        // μπορούμε να το προσθέσουμε εδώ.
        return AISCORE_BASE + "?" + params.toString();
      }

      function setActiveNavButtons() {
        const allNavBtns = document.querySelectorAll(".nav-btn");
        allNavBtns.forEach((btn) => {
          const view = btn.getAttribute("data-view");
          btn.classList.toggle("active", view === state.view);
        });
      }

      function setActiveThemeButtons() {
        themeButtons.forEach((btn) => {
          const theme = btn.getAttribute("data-theme");
          btn.classList.toggle("active", theme === state.theme);
        });
        modeThemeLabel.textContent = themeLabelName(state.theme);
      }

      function setViewMetadata() {
        modeLabel.textContent = state.view.toUpperCase();

        if (state.view === "live") {
          widgetTitleLabel.textContent = "Live Matches";
          widgetSubtitleLabel.textContent =
            "Global live scores, goals, cards and minute-by-minute data via AiScore.";
        } else if (state.view === "scores") {
          widgetTitleLabel.textContent = "Scores & Schedule";
          widgetSubtitleLabel.textContent =
            "Results, upcoming fixtures and full competition tree inside the AiScore widget.";
        } else if (state.view === "standings") {
          widgetTitleLabel.textContent = "Standings & Tables";
          widgetSubtitleLabel.textContent =
            "League rankings, points, form and placement across all major competitions.";
        } else if (state.view === "leagues") {
          widgetTitleLabel.textContent = "Leagues Overview";
          widgetSubtitleLabel.textContent =
            "Browse global leagues and cups from within the AiScore widget.";
        } else if (state.view === "teams") {
          widgetTitleLabel.textContent = "Teams & Clubs";
          widgetSubtitleLabel.textContent =
            "Team pages, squads and historical performance from AiScore data.";
        } else if (state.view === "ai") {
          widgetTitleLabel.textContent = "AI Predictions Lab";
          widgetSubtitleLabel.textContent =
            "Planned module for xG, probability curves and automated edge detection.";
        } else if (state.view === "goalmatrix") {
          widgetTitleLabel.textContent = "GoalMatrix Engine";
          widgetSubtitleLabel.textContent =
            "Live goal probability overlays across time buckets and game states.";
        } else if (state.view === "smartmoney") {
          widgetTitleLabel.textContent = "SmartMoney Monitor";
          widgetSubtitleLabel.textContent =
            "Market movement tracking and bankroll exposure analytics (future).";
        } else if (state.view === "about") {
          widgetTitleLabel.textContent = "About AI MATCHLAB";
          widgetSubtitleLabel.textContent =
            "Context, roadmap and positioning of the AI-powered football lab.";
        }
      }

      function showIframe(show) {
        iframeWrapper.style.display = show ? "block" : "none";
      }

      function showPane(paneId) {
        panes.forEach((p) => p.classList.remove("active"));
        if (!paneId) return;
        const pane = document.getElementById(paneId);
        if (pane) pane.classList.add("active");
      }

      function updateIframeSrc() {
        // Για όλες τις “data” views (live/scores/standings/leagues/teams)
        // χρησιμοποιούμε το ίδιο AiScore URL.
        const isDataView = ["live", "scores", "standings", "leagues", "teams"].includes(
          state.view
        );
        if (isDataView) {
          const url = buildAiscoreUrl(state.view, state.theme);
          if (iframe.src !== url) {
            iframe.src = url;
          }
        }
      }

      function updateView(nextView) {
        state.view = nextView;
        setActiveNavButtons();
        setViewMetadata();

        const isDataView = ["live", "scores", "standings", "leagues", "teams"].includes(
          state.view
        );

        if (isDataView) {
          showIframe(true);
          showPane(null);
          updateIframeSrc();
        } else {
          showIframe(false);
          if (state.view === "ai") showPane("pane-ai");
          else if (state.view === "goalmatrix") showPane("pane-goalmatrix");
          else if (state.view === "smartmoney") showPane("pane-smartmoney");
          else if (state.view === "about") showPane("pane-about");
        }
      }

      function updateTheme(nextTheme) {
        state.theme = nextTheme;
        setActiveThemeButtons();
        updateIframeSrc();
        persistTheme();
      }

      function initNav() {
        primaryNav.addEventListener("click", (e) => {
          const btn = e.target.closest("button[data-view]");
          if (!btn) return;
          const view = btn.getAttribute("data-view");
          updateView(view);
        });

        aiNav.addEventListener("click", (e) => {
          const btn = e.target.closest("button[data-view]");
          if (!btn) return;
          const view = btn.getAttribute("data-view");
          updateView(view);
        });

        themeButtons.forEach((btn) => {
          btn.addEventListener("click", () => {
            const theme = btn.getAttribute("data-theme");
            updateTheme(theme);
          });
        });
      }

      function bootstrap() {
        loadPersistedTheme();
        setActiveThemeButtons();
        setActiveNavButtons();
        setViewMetadata();
        updateIframeSrc();
        initNav();
      }

      bootstrap();
    })();
  </script>
